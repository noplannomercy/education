**Step 1: renderTasks 함수 전체 교체**

기존 `renderTasks()` 전체를 아래로 교체:

```javascript
function renderTasks() {
  const project = state.projects.find(p => p.id === state.selectedProjectId);
  const titleEl = document.getElementById('taskPanelTitle');
  const addForm = document.getElementById('taskAddForm');
  const board = document.getElementById('kanbanBoard');
  const noMsg = document.getElementById('noProjectMsg');

  if (!project) {
    titleEl.textContent = '프로젝트를 선택하세요';
    addForm.style.display = 'none';
    board.style.display = 'none';
    noMsg.style.display = 'flex';
    return;
  }

  titleEl.textContent = project.name;
  addForm.style.display = 'flex';
  board.style.display = 'grid';
  noMsg.style.display = 'none';

  const statuses = ['todo', 'inprogress', 'done'];
  statuses.forEach(status => {
    const col = document.getElementById('col-' + status);
    const tasks = project.tasks.filter(t => t.status === status);
    document.getElementById('count-' + status).textContent = tasks.length;

    if (tasks.length === 0) {
      col.innerHTML = '<div class="col-empty">태스크 없음</div>';
    } else {
      col.innerHTML = tasks.map(t => `
        <div class="task-card" data-task-id="${t.id}" draggable="true">
          <span class="task-title" ondblclick="startEditTask('${t.id}')">${escHtml(t.title)}</span>
          <button class="btn-delete" onclick="deleteTask('${t.id}')" title="삭제">✕</button>
        </div>
      `).join('');
    }

    // D&D: dragover / dragleave / drop
    col.ondragover = e => { e.preventDefault(); col.classList.add('drag-over'); };
    col.ondragleave = e => { if (!col.contains(e.relatedTarget)) col.classList.remove('drag-over'); };
    col.ondrop = e => {
      e.preventDefault();
      col.classList.remove('drag-over');
      const taskId = e.dataTransfer.getData('text/plain');
      const newStatus = col.dataset.status;
      const proj = state.projects.find(p => p.id === state.selectedProjectId);
      const task = proj?.tasks.find(t => t.id === taskId);
      if (task && task.status !== newStatus) {
        task.status = newStatus;
        saveToStorage();
        renderProjects();
        renderTasks();
      }
    };

    // D&D: dragstart on cards
    col.querySelectorAll('.task-card').forEach(card => {
      card.ondragstart = e => {
        e.dataTransfer.setData('text/plain', card.dataset.taskId);
        card.classList.add('dragging');
      };
      card.ondragend = () => card.classList.remove('dragging');
    });
  });
}
```
